SELECT 
    c.id AS campaign_id,
    c.name AS campaign_name,
    SUM(clr.cost) AS total_cost,
    SUM(clr.clicks) AS total_clicks,
    CASE 
        WHEN SUM(clr.clicks) > 0 THEN SUM(clr.cost) / SUM(clr.clicks)
        ELSE 0 
    END AS avg_cost_per_click
FROM 
    campaign_history_data c
JOIN 
    campaign_level_report_data clr ON c.id = clr.campaign_id
GROUP BY 
    c.id, c.name
ORDER BY 
    avg_cost_per_click DESC
